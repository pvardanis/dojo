    ```python
    # ABOUTME: SQL adapter for the SourceRepository Protocol.
    # ABOUTME: Thin glue: mapper + session.merge / session.get.
    """Sync SQL implementation of SourceRepository."""

    from __future__ import annotations

    from sqlalchemy.orm import Session

    from app.domain.entities import Source
    from app.domain.value_objects import SourceId
    from app.infrastructure.db.mappers import (
        source_from_row,
        source_to_row,
    )
    from app.infrastructure.db.models import SourceRow
    from app.logging_config import get_logger

    log = get_logger(__name__)


    class SqlSourceRepository:
        """Sync SQL adapter for SourceRepository Protocol."""

        def __init__(self, session: Session) -> None:
            """Hold the request-scoped session (CONTEXT D-01b).

            :param session: SQLAlchemy Session open for the caller's
                unit of work; the repo never commits or rolls back.
            """
            self._session = session

        def save(self, source: Source) -> None:
            """Upsert the source row keyed by `source.id`.

            :param source: The `Source` entity to persist.
            """
            row = source_to_row(source)
            self._session.merge(row)
            self._session.flush()

        def get(self, source_id: SourceId) -> Source | None:
            """Return the stored source or `None` if absent.

            :param source_id: Typed id of the source to load.
            :returns: The `Source` entity, or `None` when no row
                exists for `source_id`.
            """
            row = self._session.get(SourceRow, str(source_id))
            return source_from_row(row) if row is not None else None
    ```

    ```python
    def test_third_save_failure_rolls_back_all_three(
        session: Session,
    ) -> None:
        """SC #2: forced card-save fail → Source + Note + Card absent."""
        src_repo = SqlSourceRepository(session)
        note_repo = SqlNoteRepository(session)
        card_repo = SqlCardRepository(session)

        src = Source(
            kind=SourceKind.TOPIC,
            user_prompt="p",
            display_name="d",
        )
        note = Note(
            source_id=src.id,
            title="t",
            content_md="c",
        )
        # Craft a card that collides with note.id (PK violation on
        # cards table? — use a different forced-failure path:
        # save two identical card rows in the same transaction will
        # collide on PK because session.add queues both INSERTs and
        # the second one errors at flush.
        duplicate_id = CardId(uuid.uuid4())
        card_a = Card(
            source_id=src.id,
            question="q",
            answer="a",
            id=duplicate_id,
        )
        card_b = Card(
            source_id=src.id,
            question="q2",
            answer="a2",
            id=duplicate_id,   # same id → flush-time PK violation
        )

        with pytest.raises(IntegrityError):
            with session.begin_nested():
                src_repo.save(src)
                note_repo.save(note)
                card_repo.save(card_a)
                card_repo.save(card_b)   # PK collision → raises

        # After rollback of the nested SAVEPOINT, none of the rows
        # persisted. The outer conftest transaction still wraps us;
        # use session.get on the row types directly.
        assert session.get(SourceRow, str(src.id)) is None
        assert session.get(NoteRow, str(note.id)) is None
        assert session.get(CardRow, str(duplicate_id)) is None
    ```

    Notes on the pattern:
    - We use `session.begin_nested()` instead of `session.begin()`
      because the conftest `session` fixture is already inside an outer
      transaction (opened by the fixture on the connection, line
      99-100). `begin_nested` opens a SAVEPOINT under that outer
      transaction; a raise inside rolls back the SAVEPOINT but leaves
      the outer transaction alive so `session.get` calls after the
      exception still work (the fixture does `outer.rollback()` at
      teardown, fully discarding the test's writes).
    - This tests the SAME semantics as Phase 4's `with session.begin():`
      — SQLAlchemy makes `begin` and `begin_nested` isomorphic in terms
      of rollback-on-exception behavior. Document this in the test
      docstring.
    - Force-failure mechanism: duplicate primary key on the second
      `card_repo.save(card_b)` triggers an `IntegrityError` at flush.
      Deterministic; no constraint hacking needed. `flush()` in the
      repo's save method surfaces the error synchronously in the `with`
      block (not lazily at commit).
