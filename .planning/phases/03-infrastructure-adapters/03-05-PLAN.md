    ```python
    # ABOUTME: In-memory DraftStore — plain dict + lazy TTL + atomic pop.
    # ABOUTME: GIL-atomic dict.pop is the load-bearing concurrency claim.
    """InMemoryDraftStore — CPython GIL-atomic dict-backed DraftStore."""

    from __future__ import annotations

    import time
    from collections.abc import Callable

    from app.application.dtos import DraftBundle
    from app.application.ports import DraftToken
    from app.logging_config import get_logger

    log = get_logger(__name__)

    _TTL_SECONDS = 30 * 60


    class InMemoryDraftStore:
        """DraftStore adapter: plain dict + lazy TTL + GIL-atomic pop.

        Thread safety comes from CPython GIL atomicity of ``dict.pop``.
        If Dojo ever runs on a no-GIL Python build (PEP 703), swap in
        ``threading.Lock`` — the free-threaded build's per-dict locking
        prevents corruption but does not document "exactly one caller
        wins" semantics for a concurrent race (CONTEXT D-04c).
        """

        def __init__(
            self,
            clock: Callable[[], float] = time.monotonic,
            ttl_seconds: float = _TTL_SECONDS,
        ) -> None:
            """Build the store with an injected clock + TTL window.

            :param clock: Zero-arg callable returning a monotonic float.
                Tests inject a list-captured counter for fake-clock TTL
                coverage.
            :param ttl_seconds: Eviction window; defaults to 30 minutes
                (DRAFT-01).
            """
            self._store: dict[
                DraftToken, tuple[DraftBundle, float]
            ] = {}
            self._clock = clock
            self._ttl = ttl_seconds

        def put(self, token: DraftToken, bundle: DraftBundle) -> None:
            """Store the bundle with a current-time stamp.

            :param token: Token minted by the use case.
            :param bundle: Draft to hold until save or discard.
            """
            self._store[token] = (bundle, self._clock())

        def pop(self, token: DraftToken) -> DraftBundle | None:
            """Atomic read-and-delete; None if missing or expired.

            :param token: Draft token to consume.
            :returns: The bundle, or None when absent or past TTL.
            """
            entry = self._store.pop(token, None)
            if entry is None:
                return None
            bundle, stored_at = entry
            if self._clock() - stored_at > self._ttl:
                return None
            return bundle
    ```

    ```python
    # ABOUTME: FILE source reader — UTF-8 strict, wraps OSError family.
    # ABOUTME: Module-level function (Callable[[Path], str] port).
    """FILE source reader — SourceReader Protocol impl."""

    from __future__ import annotations

    from pathlib import Path

    from app.application.exceptions import (
        SourceNotFound,
        SourceUnreadable,
    )
    from app.logging_config import get_logger

    log = get_logger(__name__)


    def read_file(path: Path) -> str:
        """Read ``path`` as strict UTF-8 text.

        :param path: Filesystem path to the source file.
        :returns: Decoded UTF-8 contents of the file.
        :raises SourceNotFound: When the path does not exist.
        :raises SourceUnreadable: On permission, encoding, directory,
            or other OS errors.
        """
        try:
            return path.read_text(encoding="utf-8", errors="strict")
        except FileNotFoundError as e:
            raise SourceNotFound(str(path)) from e
        except PermissionError as e:
            raise SourceUnreadable(f"permission denied: {path}") from e
        except UnicodeDecodeError as e:
            raise SourceUnreadable(f"not valid UTF-8: {path}") from e
        except IsADirectoryError as e:
            raise SourceUnreadable(f"path is a directory: {path}") from e
        except OSError as e:
            raise SourceUnreadable(f"OS error reading {path}: {e}") from e
    ```

    ```python
    # ABOUTME: URL fetcher — httpx sync GET + trafilatura extraction.
    # ABOUTME: Paywall/length heuristics per CONTEXT D-05.
    """URL fetcher — UrlFetcher Protocol impl."""

    from __future__ import annotations

    import httpx
    import trafilatura

    from app.application.exceptions import (
        SourceFetchFailed,
        SourceNotArticle,
    )
    from app.logging_config import get_logger

    log = get_logger(__name__)

    _USER_AGENT = "Dojo/0.1 (+local study app)"
    _TIMEOUT = httpx.Timeout(connect=5.0, read=10.0, write=5.0, pool=5.0)
    _MIN_CHARS = 1000
    _PAYWALL_MAX_CHARS = 2000
    _PAYWALL_MARKERS = frozenset(
        (
            "subscribe",
            "sign up",
            "sign in",
            "paywall",
            "continue reading",
            "free article",
        )
    )


    def fetch_url(url: str) -> str:
        """Fetch ``url`` and extract main article text.

        :param url: HTTP/HTTPS URL to fetch.
        :returns: Extracted article text (>= 1000 chars).
        :raises SourceFetchFailed: On non-2xx, timeout, or transport.
        :raises SourceNotArticle: On extraction failure, too-short
            content, or paywall-marker heuristic.
        """
        try:
            response = httpx.get(
                url,
                headers={"User-Agent": _USER_AGENT},
                follow_redirects=True,
                timeout=_TIMEOUT,
            )
        except httpx.TimeoutException as e:
            raise SourceFetchFailed(f"timeout fetching {url}") from e
        except httpx.HTTPError as e:
            raise SourceFetchFailed(f"fetch failed: {e}") from e

        if not response.is_success:
            raise SourceFetchFailed(
                f"URL returned {response.status_code}"
            )

        extracted = trafilatura.extract(
            response.text,
            include_comments=False,
            deduplicate=True,
            output_format="txt",
            favor_precision=True,
        )
        if extracted is None or len(extracted) < _MIN_CHARS:
            raise SourceNotArticle("extraction failed or too short")

        if len(extracted) < _PAYWALL_MAX_CHARS:
            low = extracted.lower()
            hits = sum(1 for m in _PAYWALL_MARKERS if m in low)
            if hits >= 2:
                raise SourceNotArticle("paywall suspected")

        return extracted
    ```

    ```python
        Framing: this is a CPython 3.12 GIL observability gate (RESEARCH
    ```
